## Background

- **Task 1**: `_get_relative_path()` and `_get_module_path()` are called 6+ times per file during AST traversal but always return the same value (inputs don't change). `os.path.relpath` involves string splitting — negligible per call but measurable at 10k+ files.
- **Task 2**: `call_llm` in `llm_services.py:203` creates a fresh `OpenAI(...)` client on every call. Each new client spawns a new `httpx.Client` with a new connection pool — TLS handshake on every LLM request instead of reusing keep-alive connections.
- **Task 3**: `_RETRY_DELAYS = [10, 30, 90]` are fixed; under concurrent load all workers retry simultaneously ("thundering herd"). 429 responses may include a `Retry-After` header that's currently ignored.
- **Task 4**: `fix_docs` always re-processes all `.md` files even when they haven't changed since the last run. Phase 1 (mdformat) is purely mechanical and trivially parallel.

```python
import random


def _parse_retry_after(exc: Exception) -> float | None:
    """Extract Retry-After seconds from a 429 RateLimitError response, if present.

    Returns None for any other exception type or when the header is absent.
    """
    import openai
    if not isinstance(exc, openai.RateLimitError):
        return None
    headers = getattr(getattr(exc, "response", None), "headers", {})
    val = headers.get("retry-after") or headers.get("Retry-After")
    if val:
        try:
            return float(val)
        except ValueError:
            pass
    return None


def _sleep_with_jitter(base_delay: float) -> None:
    """Sleep for base_delay + random jitter (uniform 0–50 % of base).

    Avoids the thundering-herd problem when multiple workers all retry
    simultaneously after a rate-limit or transient server error.
    """
    actual = base_delay + random.uniform(0, base_delay * 0.5)
    time.sleep(actual)
```

```python
_FIX_CACHE_FILENAME = ".fix_docs_cache.json"


def _file_hash(text: str) -> str:
    """Return a 16-char hex digest of the text (SHA-256 truncated)."""
    return hashlib.sha256(text.encode("utf-8")).hexdigest()[:16]


def _load_hash_cache(docs_path: Path) -> dict[str, str]:
    """Load the hash cache from disk; return empty dict on any error."""
    cache_file = docs_path / _FIX_CACHE_FILENAME
    if cache_file.exists():
        try:
            return json.loads(cache_file.read_text(encoding="utf-8"))
        except Exception:
            return {}
    return {}


def _save_hash_cache(docs_path: Path, cache: dict[str, str]) -> None:
    """Persist the hash cache to disk."""
    cache_file = docs_path / _FIX_CACHE_FILENAME
    try:
        cache_file.write_text(json.dumps(cache, indent=2), encoding="utf-8")
    except Exception as exc:
        logger.warning(f"Could not save fix_docs cache: {exc}")
```

Replace the entire `fix_docs` function body with:

```python
def fix_docs(working_dir: str, config: Config) -> FixStats:
    """Apply all three post-processing fix phases to every .md file in *working_dir*.

    Phase 1 — Markdown formatting (mdformat, mechanical) — parallelised
    Phase 2 — Math repair (LaTeX validation + LLM)
    Phase 3 — Mermaid repair (mmdc/regex + LLM)

    Hash-based incremental skip: if a file's content is unchanged since the last
    successful fix run, Phases 2 and 3 are skipped for that file.
    """
    docs_path = Path(working_dir)
    md_files = sorted(docs_path.glob("*.md"))

    if not md_files:
        return FixStats()

    mmdc_available = _find_mmdc() is not None
    validation_mode = "mmdc" if mmdc_available else "regex heuristics"
    logger.info(
        f"🔧 Post-processing {len(md_files)} file(s): markdown + math + mermaid "
        f"(mermaid validation: {validation_mode})"
    )

    stats = FixStats()
    hash_cache = _load_hash_cache(docs_path)
    updated_cache: dict[str, str] = {}

    # ── Phase 1: parallel mdformat ─────────────────────────────────────────────
    def _phase1(md_file: Path) -> tuple[Path, str]:
        """Read, format, return (path, formatted_text). Thread-safe: no shared state."""
        try:
            text = md_file.read_text(encoding="utf-8")
            return md_file, _format_markdown(text, stats)
        except Exception as exc:
            logger.warning(f"  ✗ Phase 1 failed for {md_file.name}: {exc}")
            return md_file, md_file.read_text(encoding="utf-8")

    max_workers = min(32, (os.cpu_count() or 1) + 4)
    phase1_results: dict[Path, str] = {}
    with ThreadPoolExecutor(max_workers=max_workers) as executor:
        for md_file, formatted in executor.map(_phase1, md_files):
            phase1_results[md_file] = formatted

    # ── Phases 2+3: serial, with hash-based skip ───────────────────────────────
    for md_file in md_files:
        try:
            text = phase1_results[md_file]
            original_raw = md_file.read_text(encoding="utf-8")

            # Write Phase 1 result if changed
            if text != original_raw:
                md_file.write_text(text, encoding="utf-8")

            # Check incremental cache: skip LLM phases if content unchanged
            cache_key = md_file.name
            current_hash = _file_hash(text)
            if hash_cache.get(cache_key) == current_hash:
                updated_cache[cache_key] = current_hash
                continue  # skip Phases 2+3

            # Phase 2 — Math repair
            text = _fix_math_in_text(text, config, stats)

            # Phase 3 — Mermaid repair
            text = _fix_mermaid_in_text(text, config, stats)

            if text != original_raw:
                md_file.write_text(text, encoding="utf-8")

            # Store hash of final output so next run can skip
            updated_cache[cache_key] = _file_hash(text)

        except Exception as exc:
            logger.warning(f"  ✗ Failed to process {md_file.name}: {exc}")

    _save_hash_cache(docs_path, updated_cache)

    if stats.md_files_formatted:
        logger.info(f"  📝 Formatted {stats.md_files_formatted} file(s) with mdformat")
    if stats.math_repaired:
        logger.info(f"  ✓ Math: repaired {stats.math_repaired}/{stats.math_invalid} formula(s)")
    if stats.diagrams_repaired:
        logger.info(f"  ✓ Mermaid: repaired {stats.diagrams_repaired}/{stats.diagrams_invalid} diagram(s)")

    return stats
```


The `stats` object is shared across threads via `_phase1` closures. `FixStats` uses simple integer attributes (`+=`). Integer increment is **not** atomic in Python under concurrent writes. Fix: don't pass `stats` into `_phase1`; instead collect a per-file stats delta and merge in the main thread:

```python
def _phase1(md_file: Path) -> tuple[Path, str, int]:
    """Returns (path, formatted_text, files_formatted_delta)."""
    try:
        text = md_file.read_text(encoding="utf-8")
        local_stats = FixStats()
        formatted = _format_markdown(text, local_stats)
        return md_file, formatted, local_stats.md_files_formatted
    except Exception as exc:
        logger.warning(f"  ✗ Phase 1 failed for {md_file.name}: {exc}")
        return md_file, md_file.read_text(encoding="utf-8"), 0

# In the main loop:
with ThreadPoolExecutor(max_workers=max_workers) as executor:
    for md_file, formatted, delta in executor.map(_phase1, md_files):
        phase1_results[md_file] = formatted
        stats.md_files_formatted += delta  # safe: single-threaded merge
```
